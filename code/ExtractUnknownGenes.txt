#These three commands will generate all of our "unknown" genes into a txt file. replace output with the desired output file and input with the annotation_info.txt file for the genome 

1)
Get-Content "INPUT" | Where-Object {$_ -match 'unknown|DUF|putative|uncharacterised' } | Out-File -FilePath "OUTPUT"

2)
Get-Content "INPUT" | ForEach-Object {
    $columns = $_ -split "`t"
    if ($columns[12] -eq "") {
        $_
    }
} | Out-File -FilePath "OUTPUT" -Append

3)
Get-Content "INPUT" | ForEach-Object {
    $columns = $_ -split "`t"  
    if ($columns[12] -eq ".") {
	$_
    }
} | Out-File -FilePath "OUTPUT" -Append
